Create app using qsocks

        qsocks.Connect()
.then( (global) => global.createDocEx('Qonnections Demo') )
.then( (app) => {
    
    const loadscript = 
    `Load RecNo() as Dim, Rand() * 100 as Value
     Autogenerate 100;`
    
    return app.setScript(loadscript)
    .then(() => app.doReload())
    .then(() => app.doSave())

})
